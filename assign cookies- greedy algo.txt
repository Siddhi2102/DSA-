class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        sort(g.begin(),g.end());
        sort(s.begin(),s.end());
        int l=0,r=0;
        int cnt=0;
        while(r<s.size() && l<g.size()){
            if(g[l]<=s[r]){
                l=l+1;
                cnt++;
            }
            r=r+1;

        }

        return cnt;


        
    }
};
